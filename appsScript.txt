function doGet(e) {
  var func = e.parameter.function;

  if (func == "getMovies") {
    return getMovies();
  } else {
    return ContentService.createTextOutput(`Função ${func} não encontrada`);
  }
}

function doPost(e) {
  var func = e.parameter.function;

  switch(func) {
    case 'postMovie':
      var requestBody = e.postData.contents;
      var movieData = JSON.parse(requestBody);
      return postMovie(e, movieData);
    case 'deleteMovie':
      return deleteMovie(e);
    default:
      return ContentService.createTextOutput(`Função ${func} não encontrada`);
  }
}

function getMovies() {
  var sheet = SpreadsheetApp.openById("1JClY82U2bNpZd8pyOmPGnynUJbWEB_-0NbKKV5QK7CA");
  var page = sheet.getSheetByName("Página 1");
  var imgPage = sheet.getSheetByName("Imagens");
  var values = page.getDataRange().getValues();
  var imgValues = imgPage.getDataRange().getValues();
  var data = [];

  // começando em 1, pois a primeira coluna é apenas o id de cada página.
  for (var i = 1; i < values.length; i ++) {
    var row = values[i];
    var imgRow = imgValues[i];
    var feedback = {};
    feedback['id'] = row[0];
    feedback['titulo'] = row[1].toString();
    feedback['diretor'] = row[2].toString();
    feedback['sinopse'] = row[3].toString();
    feedback['img'] = imgRow.slice(1).join("").toString();

    data.push(feedback);
  }

  return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON);
}

function postMovie (e, data) {
  var sheet = SpreadsheetApp.openById("1JClY82U2bNpZd8pyOmPGnynUJbWEB_-0NbKKV5QK7CA");
  var page = sheet.getSheetByName("Página 1");
  var imgPage = sheet.getSheetByName("Imagens");

  // elementos passados com parâmetros na url
  var movieId = parseInt(e.parameter.index);
  var rowToPost = movieId + 1;

  // elementos passados no bory da requisição
  var titulo = data.titulo;
  var diretor = data.diretor;
  var sinopse = data.sinopse;
  var img = data.img;

  var imgRow = [movieId];
  for(let i = 0; i < img.length; i += 49999) {
    imgRow.push(img.substring(i, i + 49999));
  }

  // range(linha, coluna, numLinhas, numColunas)
  var range = page.getRange(rowToPost, 1, 1, 5);
  var imgPageRange = imgPage.getRange(rowToPost, 1, 1, imgRow.length);

  range.setValues([[movieId, titulo, diretor, sinopse, ""]]);
  imgPageRange.setValues([imgRow]);

  return ContentService.createTextOutput(`Salvo... ${titulo}, ${diretor}, ${sinopse}, ${img}`);
}

function deleteMovie (e) {
  var sheet = SpreadsheetApp.openById("1JClY82U2bNpZd8pyOmPGnynUJbWEB_-0NbKKV5QK7CA");
  var page = sheet.getSheetByName("Página 1");
  var imgPage = sheet.getSheetByName("Imagens");

  var rowToDelete = parseInt(e.parameter.index) + 1;
  var lastRow = sheet.getLastRow();

  page.deleteRow(rowToDelete);
  imgPage.deleteRow(rowToDelete);

  for (var i = rowToDelete; i <= lastRow - 1; i++) {
    page.getRange(i, 1).setValue(i - 1);
    imgPage.getRange(i, 1).setValue(i - 1);
  }

  return ContentService.createTextOutput("Deletado...");
}

function createBackup() {
  var spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
  var spreadsheetName = spreadsheet.getName();
  var backupName = spreadsheetName + " - " + Utilities.formatDate(new Date(), Session.getTimeZone(), "yyyy-MM-dd HH:mm:ss");
  var backupFolder = DriveApp.getFolderById("1AvP5hubSLvCrixlZHh0XLmzEpl6BFIDy");

  // Copia planilha e salva na pasta de backup
  var backupCopy = spreadsheet.copy(backupName + " - Backup");
  var backupId = backupCopy.getId();
  var backupFile = DriveApp.getFileById(backupId);
  backupFolder.addFile(backupFile);

  // Manda um email confirmando a criação do backup
  var recipient = Session.getActiveUser().getEmail();
  var subject = "Backup criado para a planilha " + spreadsheetName;
  var message = "Foi criado um backup da planilha " + spreadsheetName + " com o nome " + backupName + "- Backup.";
  MailApp.sendEmail(recipient, subject, message);
}
